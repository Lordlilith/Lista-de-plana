\section*{Quest\~ao 1}
\addcontentsline{toc}{section}{Quest\~ao 1}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\textbf{Configura\c{c}\~ao:} Os pontos $A, B, C, D$ est\~ao sobre uma reta, nessa ordem da esquerda para a direita. Usamos a nota\c{c}\~ao: $\overline{AB}$ para o segmento de extremos $A$ e $B$, e $S_{AB}$ para a semirreta de origem $A$ passando por $B$.

\bigskip
\textbf{(a)} $\overline{AB} \cup \overline{BC}$

Como $B$ est\'a entre $A$ e $C$, todo ponto de $\overline{AB}$ ou de $\overline{BC}$ satisfaz $A \leq X \leq C$. Logo:
$$\overline{AB} \cup \overline{BC} = \overline{AC}$$

\textbf{(b)} $\overline{AB} \cap \overline{BC}$

$\overline{AB}$: $A \leq X \leq B$; $\overline{BC}$: $B \leq X \leq C$. O \'unico valor satisfazendo $X \leq B$ e $X \geq B$ \'e $X = B$. Logo:
$$\overline{AB} \cap \overline{BC} = \{B\}$$

\textbf{(c)} $\overline{AC} \cap \overline{BD}$

$\overline{AC}$: $A \leq X \leq C$; $\overline{BD}$: $B \leq X \leq D$. A interse\c{c}\~ao exige $B \leq X \leq C$. Logo:
$$\overline{AC} \cap \overline{BD} = \overline{BC}$$

\textbf{(d)} $\overline{AB} \cap \overline{CD}$

$\overline{AB}$: $A \leq X \leq B$; $\overline{CD}$: $C \leq X \leq D$. Como $B < C$, n\~ao existe $X$ satisfazendo ambas simultaneamente. Logo:
$$\overline{AB} \cap \overline{CD} = \emptyset$$

\textbf{(e)} $S_{AB} \cap S_{BC}$

$S_{AB}$: todos os $X \geq A$; $S_{BC}$: todos os $X \geq B$. Como $A < B$, temos $S_{BC} \subset S_{AB}$. Logo:
$$S_{AB} \cap S_{BC} = S_{BC}$$

\textbf{(f)} $S_{AB} \cap S_{AD}$

Ambas as semirretas t\^em origem $A$ e o mesmo sentido (para a direita), portanto s\~ao id\^enticas:
$$S_{AB} \cap S_{AD} = S_{AB} = S_{AD}$$

\textbf{(g)} $S_{CB} \cap S_{BC}$

$S_{CB}$: todos os $X \leq C$; $S_{BC}$: todos os $X \geq B$. A interse\c{c}\~ao exige $B \leq X \leq C$. Logo:
$$S_{CB} \cap S_{BC} = \overline{BC}$$

\textbf{(h)} $S_{AB} \cup S_{BC}$

$S_{AB}$: todos os $X \geq A$; $S_{BC}$: todos os $X \geq B$. Como $S_{BC} \subset S_{AB}$:
$$S_{AB} \cup S_{BC} = S_{AB}$$

\section*{Quest\~ao 3}
\addcontentsline{toc}{section}{Quest\~ao 3}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\textbf{Enunciado (item (b) da Proposi\c{c}\~ao 1.4):} $S_{AB} \cap S_{BA} = \overline{AB}$.

\begin{proof}
Seja $r$ a reta com coordenadas tais que $A$ tem coordenada $a$ e $B$ tem coordenada $b$, com $a < b$. Por defini\c{c}\~ao:
$$S_{AB} = \{X \in r : x \geq a\}, \qquad S_{BA} = \{X \in r : x \leq b\}$$

$(\subseteq)$: Se $X \in S_{AB} \cap S_{BA}$, ent\~ao $x \geq a$ e $x \leq b$, logo $a \leq x \leq b$, portanto $X \in \overline{AB}$.

$(\supseteq)$: Se $X \in \overline{AB}$, ent\~ao $a \leq x \leq b$, logo $X \in S_{AB}$ e $X \in S_{BA}$, portanto $X \in S_{AB} \cap S_{BA}$.

Conclu\'imos que $S_{AB} \cap S_{BA} = \overline{AB}$.
\end{proof}

\section*{Quest\~ao 4}
\addcontentsline{toc}{section}{Quest\~ao 4}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\textbf{Enunciado:} Prove que existe uma infinidade de pontos em um segmento.

\begin{proof}
Seja $\overline{AB}$ com $A \neq B$. Fixemos coordenadas tal que $A = 0$ e $B = 1$ (Axioma da R\'egua). Para cada $n \in \mathbb{N}$, $n \geq 1$, seja $P_n$ o ponto de coordenada
$$x_n = \frac{1}{n+1}$$
Como $x_n \in (0,1)$, cada $P_n \in \overline{AB}$. Al\'em disso, se $P_m = P_n$ ent\~ao $\frac{1}{m+1} = \frac{1}{n+1}$, logo $m = n$. Portanto os pontos $P_n$ s\~ao dois a dois distintos, e $\overline{AB}$ cont\'em infinitos pontos.
\end{proof}

\section*{Quest\~ao 6}
\addcontentsline{toc}{section}{Quest\~ao 6}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\textbf{Enunciado:} Mostre que a interse\c{c}\~ao de dois semiplanos \'e convexa.

\begin{proof}
Sejam $H_1$ e $H_2$ dois semiplanos e $K = H_1 \cap H_2$. Tomemos $X, Y \in K$ quaisquer. Como $X, Y \in H_1$ e $H_1$ \'e convexo:
$$\overline{XY} \subset H_1$$
Como $X, Y \in H_2$ e $H_2$ \'e convexo:
$$\overline{XY} \subset H_2$$
Portanto $\overline{XY} \subset H_1 \cap H_2 = K$. Como $X$ e $Y$ foram arbitr\'arios, $K$ \'e convexo.
\end{proof}

\section*{Quest\~ao 7}
\addcontentsline{toc}{section}{Quest\~ao 7}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\textbf{Enunciado:} Mostre que a interse\c{c}\~ao de $n$ semiplanos \'e convexa.

\begin{proof}
Por indu\c{c}\~ao sobre $n$.

\textbf{Base ($n=1$):} Um semiplano \'e convexo por hip\'otese.

\textbf{Passo indutivo:} Suponha que a interse\c{c}\~ao de quaisquer $n-1$ semiplanos seja convexa. Dados $H_1, \ldots, H_n$, seja $K_{n-1} = H_1 \cap \cdots \cap H_{n-1}$, que \'e convexo pela hip\'otese de indu\c{c}\~ao. Ent\~ao
$$\bigcap_{i=1}^n H_i = K_{n-1} \cap H_n$$
\'e a interse\c{c}\~ao de dois convexos, que \'e convexa pela Quest\~ao 6. Por indu\c{c}\~ao, o resultado vale para todo $n \geq 1$.
\end{proof}

\section*{Quest\~ao 12}
\addcontentsline{toc}{section}{Quest\~ao 12}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\textbf{Enunciado:} Prove que, se uma reta intercepta um lado de um tri\^angulo e n\~ao passa por nenhum v\'ertice, ent\~ao ela intercepta tamb\'em um dos outros dois lados. (\textbf{Teorema de Pasch})

\begin{proof}
Sejam $A$, $B$, $C$ os v\'ertices de $\triangle ABC$ e $\ell$ uma reta que intercepta $\overline{AB}$ em $P$ (com $P \neq A$, $P \neq B$) sem passar por nenhum v\'ertice.

A reta $\ell$ divide o plano em dois semiplanos $H^+$ e $H^-$. Como $P \in \overline{AB} \cap \ell$, os pontos $A$ e $B$ est\~ao em semiplanos opostos:
$$A \in H^+, \quad B \in H^-$$

Como $\ell$ n\~ao passa por $C$, temos dois casos:

\textbf{Caso 1: $C \in H^+$.} Ent\~ao $A, C \in H^+$, logo $\overline{AC}$ n\~ao cruza $\ell$. Como $B \in H^-$ e $C \in H^+$, o segmento $\overline{BC}$ cruza $\ell$, interceptando o lado $\overline{BC}$.

\textbf{Caso 2: $C \in H^-.$.} Ent\~ao $B, C \in H^-$, logo $\overline{BC}$ n\~ao cruza $\ell$. Como $A \in H^+$ e $C \in H^-$, o segmento $\overline{AC}$ cruza $\ell$, interceptando o lado $\overline{AC}$.

Em ambos os casos, $\ell$ intercepta exatamente um dos outros dois lados.
\end{proof}

\section*{Quest\~ao 14}
\addcontentsline{toc}{section}{Quest\~ao 14}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\textbf{Enunciado:} Mostre que n\~ao existe geometria com 6 pontos em que valem I1, I2 e todas as retas t\^em exatamente 3 pontos.

\begin{proof}
Suponha por contradi\c{c}\~ao que tal geometria exista. Pelo Axioma I2, cada par de pontos determina uma reta. Como cada reta cont\'em $\binom{3}{2} = 3$ pares, e o total de pares \'e $\binom{6}{2} = 15$, o n\'umero de retas seria:
$$r = \frac{15}{3} = 5$$

O n\'umero total de incid\^encias (pares ponto-reta) \'e $5 \times 3 = 15$. Dividindo pelos 6 pontos, cada ponto pertenceria a:
$$k = \frac{15}{6} = 2{,}5$$
retas. Como $k$ n\~ao \'e inteiro, temos uma contradi\c{c}\~ao. Logo, tal geometria n\~ao existe.
\end{proof}

\section*{Quest\~ao 15}
\addcontentsline{toc}{section}{Quest\~ao 15}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\textbf{Enunciado:} Se $C \in S_{AB}$ e $C \neq A$, mostre que $S_{AB} = S_{AC}$, que $\overline{BC} \subset S_{AB}$ e que $A \notin \overline{BC}$.

\textbf{Configura\c{c}\~ao:} Coordenadas com $A = 0$, $B = b > 0$, $C = c > 0$ (pois $C \in S_{AB}$ e $C \neq A$).

\begin{proof}[Prova de $S_{AB} = S_{AC}$]
$S_{AB} = \{X : x \geq 0\}$ e $S_{AC} = \{X : x \geq 0\}$, pois ambas t\^em origem $A = 0$ e sentido positivo ($b > 0$, $c > 0$). Logo $S_{AB} = S_{AC}$.
\end{proof}

\begin{proof}[Prova de $\overline{BC} \subset S_{AB}$]
Todo $X \in \overline{BC}$ tem coordenada $x \geq \min(b,c) > 0$, logo $x \geq 0$, portanto $X \in S_{AB}$.
\end{proof}

\begin{proof}[Prova de $A \notin \overline{BC}$]
$A$ tem coordenada $0$, e todo $X \in \overline{BC}$ satisfaz $x \geq \min(b,c) > 0$. Logo $A \notin \overline{BC}$.
\end{proof}

\section*{Quest\~ao 16}
\addcontentsline{toc}{section}{Quest\~ao 16}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\textbf{Enunciado:} Demonstre que a interse\c{c}\~ao de convexos \'e convexa.

\begin{proof}
Seja $\{K_\lambda\}_{\lambda \in \Lambda}$ uma fam\'ilia de conjuntos convexos e $K = \bigcap_{\lambda \in \Lambda} K_\lambda$. Tomemos $X, Y \in K$ quaisquer. Para cada $\lambda$, $X, Y \in K_\lambda$, e como $K_\lambda$ \'e convexo:
$$\overline{XY} \subset K_\lambda$$
Como isso vale para todo $\lambda$, temos $\overline{XY} \subset \bigcap_\lambda K_\lambda = K$. Logo $K$ \'e convexo.
\end{proof}

\section*{Quest\~ao 17}
\addcontentsline{toc}{section}{Quest\~ao 17}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\textbf{Enunciado:} Mostre que um tri\^angulo separa o plano em duas regi\~oes, uma das quais \'e convexa.

\begin{proof}
Seja $\triangle ABC$ com v\'ertices n\~ao colineares. Cada lado pertence a uma reta que divide o plano em dois semiplanos abertos. Denote:
\begin{align*}
H_{AB}^+ &= \text{semiplano determinado por } \ell_{AB} \text{ que cont\'em } C\\
H_{BC}^+ &= \text{semiplano determinado por } \ell_{BC} \text{ que cont\'em } A\\
H_{CA}^+ &= \text{semiplano determinado por } \ell_{CA} \text{ que cont\'em } B
\end{align*}

O interior do tri\^angulo \'e:
$$\mathrm{Int}(\triangle ABC) = H_{AB}^+ \cap H_{BC}^+ \cap H_{CA}^+$$

Pela Quest\~ao 7, interse\c{c}\~ao de semiplanos \'e convexa. Logo o interior \'e convexo. Pelo Teorema da Curva de Jordan (caso poligonal), o complementar do tri\^angulo \'e composto por duas regi\~oes: o interior (convexo) e o exterior (ilimitado, n\~ao convexo em geral).
\end{proof}

\section*{Quest\~ao 18}
\addcontentsline{toc}{section}{Quest\~ao 18}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\textbf{Generaliza\c{c}\~ao do Exerc\'icio 9} (n\'umero de retas por $n$ pontos em posi\c{c}\~ao geral):
$$r(n) = \binom{n}{2} = \frac{n(n-1)}{2}$$

\textbf{Generaliza\c{c}\~ao do Exerc\'icio 12} (Pasch para pol\'igonos convexos de $n$ lados):

Se uma reta $\ell$ n\~ao passa por nenhum v\'ertice de um pol\'igono convexo de $n$ lados e intercepta um dos seus lados, ent\~ao ela intercepta exatamente outro lado.

\begin{proof}
A reta $\ell$ divide o plano em $H^+$ e $H^-$. Os $n$ v\'ertices (nenhum sobre $\ell$) est\~ao distribu\'idos entre os dois semiplanos. Ao percorrer os v\'ertices em ordem, cada travessia de semiplano corresponde a um lado que cruza $\ell$. O n\'umero de travessias \'e par (pois retornamos ao semiplano inicial). Como h\'a pelo menos uma travessia, h\'a pelo menos duas. No caso convexo, o n\'umero \'e exatamente $\mathbf{2}$.
\end{proof}

\textbf{Parte 1:} Dois segmentos distintos podem ter dois (ou mais) pontos em comum. Exemplo: $\overline{AB} \subset \overline{AC}$ com $B$ entre $A$ e $C$ — a interse\c{c}\~ao \'e $\overline{AB}$, que cont\'em infinitos pontos.

\begin{proof}
Suponha que $\overline{PQ}$ e $\overline{RS}$ sejam distintos com $X \neq Y$ em comum. Pelo Axioma I2, ambos est\~ao na reta determinada por $X$ e $Y$, logo s\~ao colineares. Como $\overline{XY} \subset \overline{PQ}$ e $\overline{XY} \subset \overline{RS}$ (por convexidade), a interse\c{c}\~ao cont\'em todos os infinitos pontos de $\overline{XY}$. Logo a interse\c{c}\~ao nunca pode ser exatamente dois pontos.
\end{proof}

\section*{Quest\~ao 24}
\addcontentsline{toc}{section}{Quest\~ao 24}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\textbf{Hip\'otese:} $\dfrac{a}{b} = \dfrac{c}{d}$, equivalente a $ad = bc$. Seja $k = \dfrac{a}{b} = \dfrac{c}{d}$, logo $a = kb$ e $c = kd$.

\textbf{(a)} $\dfrac{a}{c} = \dfrac{b}{d}$: de $ad = bc$, dividindo por $cd$:
$$\frac{a}{c} = \frac{b}{d} \qquad \blacksquare$$

\textbf{(b)} $\dfrac{a+b}{a} = \dfrac{c+d}{c}$ e $\dfrac{a-b}{a} = \dfrac{c-d}{c}$:
$$\frac{a+b}{a} = \frac{kb+b}{kb} = \frac{k+1}{k} = \frac{kd+d}{kd} = \frac{c+d}{c} \qquad \blacksquare$$
$$\frac{a-b}{a} = \frac{kb-b}{kb} = \frac{k-1}{k} = \frac{kd-d}{kd} = \frac{c-d}{c} \qquad \blacksquare$$

\textbf{(c)} $\dfrac{a+b}{b} = \dfrac{c+d}{d}$ e $\dfrac{a-b}{b} = \dfrac{c-d}{d}$:
$$\frac{a+b}{b} = \frac{kb+b}{b} = k+1 = \frac{kd+d}{d} = \frac{c+d}{d} \qquad \blacksquare$$
$$\frac{a-b}{b} = \frac{kb-b}{b} = k-1 = \frac{kd-d}{d} = \frac{c-d}{d} \qquad \blacksquare$$

\textbf{Enunciado:} Se $P$ \'e ponto de interse\c{c}\~ao de c\'irculos de raio $r$ e centros $A$ e $B$, mostre que $PA = PB$.

\begin{proof}
$P$ pertence ao c\'irculo de centro $A$ e raio $r$, logo $m(PA) = r$. $P$ pertence ao c\'irculo de centro $B$ e raio $r$, logo $m(PB) = r$. Portanto:
$$m(PA) = r = m(PB) \implies PA = PB \qquad \blacksquare$$
\end{proof}

\textbf{Constru\c{c}\~ao de tri\^angulo isc\'osceles com r\'egua e compasso:}

\end{enumerate}

\textbf{Enunciado:} Se $a < b$, mostre que $a < \dfrac{a+b}{2} < b$.

\begin{proof}
Como $a < b$, somando $a$ a ambos os lados: $2a < a+b$, logo $a < \dfrac{a+b}{2}$.

Como $a < b$, somando $b$ a ambos os lados: $a+b < 2b$, logo $\dfrac{a+b}{2} < b$.

Portanto $a < \dfrac{a+b}{2} < b$.
\end{proof}

\textbf{Enunciado:} O c\'irculo de raio $r_1$ centrado em $A$ intercepta o c\'irculo de raio $r_2$ centrado em $B$ em exatamente dois pontos. O que se pode afirmar sobre $AB$?

Para que dois c\'irculos se intersectem em exatamente dois pontos, a dist\^ancia entre os centros deve satisfazer:
$$|r_1 - r_2| < m(AB) < r_1 + r_2$$

\textbf{Justificativa:}
\begin{itemize}
  \item Se $m(AB) = r_1 + r_2$: tangentes externamente (1 ponto).
  \item Se $m(AB) > r_1 + r_2$: disjuntos (0 pontos).
  \item Se $m(AB) = |r_1 - r_2|$: tangentes internamente (1 ponto).
  \item Se $m(AB) < |r_1 - r_2|$: um dentro do outro (0 pontos).
  \item Se $|r_1 - r_2| < m(AB) < r_1 + r_2$: exatamente 2 pontos. $\checkmark$
\end{itemize}

$$\boxed{|r_1 - r_2| < m(AB) < r_1 + r_2}$$

\textbf{Enunciado:} Dois c\'irculos de mesmo raio $r$ e centros $A$, $B$ se intersectam em $C$ e $D$.

Como $C$ e $D$ pertencem a ambos os c\'irculos: $m(AC) = m(BC) = m(AD) = m(BD) = r$.

\textbf{Sobre $\triangle ACD$:} $m(AC) = m(AD) = r$, logo \'e \textbf{isc\'osceles}.

$$m(AC) = m(CB) = m(BD) = m(DA) = r$$

\section*{Quest\~ao 34}
\addcontentsline{toc}{section}{Quest\~ao 34}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\textbf{Enunciado:} Dado $\overline{AB}$, mostre que existe e \'e \'unico $C$ entre $A$ e $B$ com $\dfrac{m(AC)}{m(BC)} = a$, $a > 0$.

\begin{proof}
Fixe coordenadas com $A = 0$ e $B = L > 0$. Seja $C$ de coordenada $x \in (0, L)$. A condi\c{c}\~ao \'e:
$$\frac{x}{L - x} = a$$

Resolvendo: $x = a(L - x) \Rightarrow x(1+a) = aL \Rightarrow x = \dfrac{aL}{1+a}$.

\textbf{Exist\^encia:} Como $a, L > 0$, temos $x > 0$. Al\'em disso, $x < L$ pois $\dfrac{a}{1+a} < 1$. Logo $C$ est\'a entre $A$ e $B$.

\textbf{Unicidade:} A equa\c{c}\~ao linear $\dfrac{x}{L-x} = a$ tem exatamente uma solu\c{c}\~ao. Logo $C$ \'e \'unico.
\end{proof}

\textbf{M\'etodo por inscri\c{c}\~ao de pol\'igonos regulares:}

Inscreva um pol\'igono regular de $n$ lados no c\'irculo de raio $r$. Cada lado tem comprimento:
$$\ell_n = 2r\sin\!\left(\frac{\pi}{n}\right)$$

O per\'imetro do pol\'igono inscrito \'e:
$$P_n = n \cdot \ell_n = 2rn\sin\!\left(\frac{\pi}{n}\right)$$

Quando $n \to \infty$:
$$\lim_{n \to \infty} P_n = \lim_{n \to \infty} 2rn\sin\!\left(\frac{\pi}{n}\right) = 2\pi r$$

\textbf{Enunciado:} O segmento ligando um ponto fora de um c\'irculo a um ponto dentro tem um ponto em comum com o c\'irculo.

\begin{proof}
Seja $\mathcal{C}$ o c\'irculo de centro $O$ e raio $r$. Sejam $P$ (fora: $m(OP) > r$) e $Q$ (dentro: $m(OQ) < r$). Defina:
$$f(t) = m(O,\, (1-t)P + tQ), \quad t \in [0,1]$$

$f$ \'e cont\'inua com $f(0) = m(OP) > r$ e $f(1) = m(OQ) < r$. Pelo \textbf{Teorema do Valor Intermedi\'ario}, existe $t^* \in (0,1)$ tal que $f(t^*) = r$. O ponto $X(t^*) \in \overline{PQ}$ pertence ao c\'irculo. $\blacksquare$
\end{proof}

\textbf{Enunciado:} Estabelecer interior e exterior da elipse $\mathcal{E} = \{C : m(CA) + m(CB) = r\}$.

\textbf{Regi\~ao interior:}
$$\mathcal{E}^{\mathrm{int}} = \{C \in \pi : m(CA) + m(CB) < r\}$$
Pontos cuja soma das dist\^ancias aos focos \'e estritamente menor que $r$. \'E um conjunto \textbf{convexo e limitado}.

\textbf{Regi\~ao exterior:}
$$\mathcal{E}^{\mathrm{ext}} = \{C \in \pi : m(CA) + m(CB) > r\}$$
Pontos cuja soma das dist\^ancias \'e estritamente maior que $r$. \'E \textbf{ilimitado e n\~ao convexo}.

O plano se decomp\~oe disjuntamente:
$$\pi = \mathcal{E}^{\mathrm{int}} \cup \mathcal{E} \cup \mathcal{E}^{\mathrm{ext}} \qquad \blacksquare$$

\section*{Quest\~ao 38}
\addcontentsline{toc}{section}{Quest\~ao 38}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\textbf{Conjuntos finitos s\~ao limitados:}

\begin{proof}
Seja $M = \{P_1, \ldots, P_n\}$ e $O$ um ponto fixo. Seja $R = \max\{m(OP_i)\} + 1$. O c\'irculo de centro $O$ e raio $R$ cont\'em todos os $P_i$. Logo $M$ \'e limitado.
\end{proof}

\textbf{Segmentos s\~ao limitados:}

\begin{proof}
Seja $O$ o ponto m\'edio de $\overline{AB}$ e $R = \dfrac{m(AB)}{2}$. Todo $X \in \overline{AB}$ satisfaz $m(OX) \leq R$. Logo $\overline{AB}$ est\'a dentro do c\'irculo de centro $O$ e raio $R$.
\end{proof}

\textbf{Tri\^angulos s\~ao limitados:}

$\triangle ABC = \overline{AB} \cup \overline{BC} \cup \overline{CA}$, uni\~ao finita de segmentos limitados. Pela Quest\~ao 39, a uni\~ao finita de limitados \'e limitada. $\blacksquare$

\section*{Quest\~ao 39}
\addcontentsline{toc}{section}{Quest\~ao 39}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\textbf{Enunciado:} Prove que a uni\~ao finita de limitados \'e limitada.

\begin{proof}
Sejam $M_1, \ldots, M_n$ conjuntos limitados e $O$ um ponto fixo. Para cada $i$, existe $\mathcal{C}_i$ de centro $O_i$ e raio $R_i$ contendo $M_i$. Para todo $P \in M_i$, pela desigualdade triangular:
$$m(OP) \leq m(OO_i) + m(O_iP) \leq m(OO_i) + R_i$$

Defina $R = \max_{1 \leq i \leq n}\{m(OO_i) + R_i\} + 1$. Ent\~ao $m(OP) \leq R$ para todo $P \in \bigcup_i M_i$. Logo a uni\~ao \'e limitada.
\end{proof}

\section*{Quest\~ao 40}
\addcontentsline{toc}{section}{Quest\~ao 40}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\textbf{Enunciado:} Dado ponto $P$ e conjunto limitado $M$, existe disco centrado em $P$ que cont\'em $M$.

\begin{proof}
Como $M$ \'e limitado, existe $\mathcal{C}_0$ de centro $O$ e raio $R_0$ com $m(OX) \leq R_0$ para todo $X \in M$. Para todo $X \in M$, pela desigualdade triangular:
$$m(PX) \leq m(PO) + m(OX) \leq m(PO) + R_0$$

Logo $M \subset D(P, R)$ com $R = m(PO) + R_0$. $\blacksquare$
\end{proof}

\section*{Quest\~ao 41}
\addcontentsline{toc}{section}{Quest\~ao 41}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\textbf{Enunciado:} Prove que as retas s\~ao conjuntos ilimitados.

\begin{proof}
Suponha por contradi\c{c}\~ao que uma reta $\ell$ seja limitada. Pela Quest\~ao 40, para qualquer $P \in \ell$, existe $R > 0$ tal que $m(PX) \leq R$ para todo $X \in \ell$.

Pelo Axioma da R\'egua, existe biject\c{c}\~ao entre os pontos de $\ell$ e $\mathbb{R}$. Fixada a coordenada $p$ de $P$, o ponto $X_n$ de coordenada $p + n$ satisfaz:
$$m(PX_n) = n$$

Para $n > R$, temos $m(PX_n) = n > R$, contrariando $m(PX_n) \leq R$. Contradi\c{c}\~ao.

Logo, toda reta \'e ilimitada. $\blacksquare$
\end{proof}
